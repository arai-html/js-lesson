script.js（要素の作成・追加）
// createElement：要素を新しく作る const li = document.createElement("li"); li.textContent = "新しいアイテム"; li.classList.add("todo-item"); // 既存の要素に追加する const ul = document.querySelector("#todo-list"); ul.appendChild(li); // 末尾に追加 ul.prepend(li); // 先頭に追加 // insertAdjacentHTML：HTML文字列で挿入（まとめて書けて便利） ul.insertAdjacentHTML("beforeend", ` <li class="todo-item"> <span>新しいタスク</span> <button class="delete-btn">削除</button> </li> `); // 要素の削除 const item = document.querySelector(".todo-item"); item.remove(); // 要素自身を削除
insertAdjacentHTMLの4つの位置
beforebegin/afterbegin/beforeend/afterend
'beforebegin'：要素の直前
'afterbegin'：要素の中・先頭
'beforeend'：要素の中・末尾 ← 最もよく使う
'afterend'：要素の直後

リストに追加するときは基本的に 'beforeend' を使います。
👆
講師メモcreateElementとinsertAdjacentHTMLは使い分けを説明。createElement→属性をJSで細かく設定したいとき。insertAdjacentHTML→HTMLをまとめて挿入するとき（テンプレートリテラルと相性が良い）。

index.html
<div class="todo-app"> <h1>ToDoリスト</h1> <div class="todo-input"> <input type="text" id="new-task" placeholder="タスクを入力..."> <button id="add-btn">追加</button> </div> <ul id="todo-list"></ul> <p id="count"></p> </div>
script.js（ToDoリスト完全版）
const input = document.querySelector("#new-task"); const addBtn = document.querySelector("#add-btn"); const list = document.querySelector("#todo-list"); const countEl = document.querySelector("#count"); let todos = []; // タスクを配列で管理 // 画面を更新する関数 const render = () => { list.innerHTML = todos.map((todo, i) => ` <li class="todo-item ${{todo.done ? "done" : ""}}"> <span>${{todo.text}}</span> <button class="done-btn" data-index="${{i}}"> ${{todo.done ? "戻す" : "完了"}} </button> <button class="del-btn" data-index="${{i}}">削除</button> </li> `).join(""); const remaining = todos.filter(t => !t.done).length; countEl.textContent = `残り ${{remaining}} タスク`; }; // タスクを追加 const addTodo = () => { const text = input.value.trim(); if (!text) return; todos.push({ text, done: false }); input.value = ""; render(); }; addBtn.addEventListener("click", addTodo); input.addEventListener("keydown", e => { if (e.key === "Enter") addTodo(); }); // 完了・削除（イベントデリゲーション） list.addEventListener("click", (e) => { const i = Number(e.target.dataset.index); if (e.target.classList.contains("done-btn")) { todos[i].done = !todos[i].done; } if (e.target.classList.contains("del-btn")) { todos.splice(i, 1); } render(); });
💡
データ（todos配列）と画面（render関数）を分けて考えるのがこのコードのポイントです。データを変えたら必ずrender()を呼ぶ——この設計パターンはReact・Vue・Angularなどモダンなフレームワークの考え方の基礎になっています。
