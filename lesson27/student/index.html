script.js（データ設計・CRUD）
// タスクのデータ構造 // { id, text, done, priority, createdAt } // localStorageから初期データを読み込む let todos = JSON.parse(localStorage.getItem("todos-v2")) || []; let filter = "all"; // all / active / done // 保存 const save = () => localStorage.setItem("todos-v2", JSON.stringify(todos)); // 追加 const addTodo = (text, priority = "medium") => { todos.unshift({ id: Date.now().toString(), text, done: false, priority, createdAt: new Date().toLocaleDateString("ja-JP") }); save(); render(); }; // 完了切り替え const toggleTodo = (id) => { todos = todos.map(t => t.id === id ? { ...t, done: !t.done } : t); save(); render(); }; // 編集 const editTodo = (id, newText) => { if (!newText.trim()) { deleteTodo(id); return; } todos = todos.map(t => t.id === id ? { ...t, text: newText.trim() } : t); save(); render(); }; // 削除 const deleteTodo = (id) => { todos = todos.filter(t => t.id !== id); save(); render(); }; // 完了済みを一括削除 const clearDone = () => { todos = todos.filter(t => !t.done); save(); render(); };
IDにDate.now()を使う理由
ユニークなIDを簡単に生成できる
Date.now()はミリ秒単位のUnixタイムスタンプを返します。同一ミリ秒以内に追加することはほぼないため、シンプルなユニークIDとして使えます。

より厳密なIDが必要な場合は crypto.randomUUID()（モダンブラウザ対応）を使います。
本番DBではサーバーサイドでIDを振ることがほとんどです。

script.js（render関数・フィルター）
// フィルター済みリストを取得 const getFiltered = () => { if (filter === "active") return todos.filter(t => !t.done); if (filter === "done") return todos.filter(t => t.done); return todos; }; // 優先度ラベルとカラー const PRIORITY = { high: { label: "高", color: "#e74c3c" }, medium: { label: "中", color: "#f39c12" }, low: { label: "低", color: "#27ae60" }, }; // render関数 const render = () => { const list = document.querySelector("#todo-list"); const countEl = document.querySelector("#todo-count"); const filtered = getFiltered(); const remaining = todos.filter(t => !t.done).length; countEl.textContent = `残り ${{remaining}} タスク`; if (filtered.length === 0) { list.innerHTML = `<li class="empty">タスクがありません</li>`; return; } list.innerHTML = filtered.map(todo => { const p = PRIORITY[todo.priority] || PRIORITY.medium; return ` <li class="todo-item ${{todo.done ? "done" : ""}}" data-id="${{todo.id}}"> <input type="checkbox" class="todo-check" ${{todo.done ? "checked" : ""}}> <span class="todo-text">${{todo.text}}</span> <span class="priority-badge" style="background:${{p.color}}">${{p.label}}</span> <button class="edit-btn">✏️</button> <button class="del-btn">🗑️</button> </li>`; }).join(""); }; // イベントデリゲーションで全イベントを処理 document.querySelector("#todo-list").addEventListener("click", (e) => { const id = e.target.closest(".todo-item")?.dataset.id; if (!id) return; if (e.target.classList.contains("todo-check")) toggleTodo(id); if (e.target.classList.contains("del-btn")) deleteTodo(id); if (e.target.classList.contains("edit-btn")) startEdit(id); });
💡
e.target.closest('.todo-item') はクリックされた要素から「最も近い祖先要素」を探すメソッドです。ボタンやチェックボックスをクリックしても親の.todo-itemを取得できるので、イベントデリゲーションでdata-idを確実に取得するための重要テクニックです。
