data.js（データ管理モジュール）
// ストレージキー const STORAGE_KEY = "reading-log-v1"; // データを読み込む const loadBooks = () => JSON.parse(localStorage.getItem(STORAGE_KEY)) || []; // データを保存する const saveBooks = (books) => localStorage.setItem(STORAGE_KEY, JSON.stringify(books)); // 書籍を追加 const addBook = (data) => { const books = loadBooks(); const newBook = { id: crypto.randomUUID(), title: data.title.trim(), author: data.author.trim(), genre: data.genre, rating: Number(data.rating) || 0, memo: data.memo.trim(), createdAt: new Date().toISOString(), }; saveBooks([newBook, ...books]); return newBook; }; // 書籍を削除 const deleteBook = (id) => { const books = loadBooks().filter(b => b.id !== id); saveBooks(books); }; // フィルター・ソート・検索を適用 const getFilteredBooks = ({ query = "", genre = "all", sort = "newest" }) => { let books = loadBooks(); if (query) { const q = query.toLowerCase(); books = books.filter(b => b.title.toLowerCase().includes(q) || b.author.toLowerCase().includes(q) ); } if (genre !== "all") books = books.filter(b => b.genre === genre); if (sort === "rating") books.sort((a, b) => b.rating - a.rating); if (sort === "title") books.sort((a, b) => a.title.localeCompare(b.title, "ja")); return books; }; // 統計情報 const getStats = () => { const books = loadBooks(); const total = books.length; const avg = total ? (books.reduce((s, b) => s + b.rating, 0) / total).toFixed(1) : "0.0"; return { total, avg }; };
💡
crypto.randomUUID()はブラウザ標準のUUID生成APIです（Chrome/Edge/Firefox/Safari対応）。Date.now()より衝突のリスクが低く、よりユニークなIDが生成できます。本番でも安心して使えます。
