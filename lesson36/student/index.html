UI描画関数を実装しよう
0〜25分
📖
ui.jsにrender系関数を実装し、データが変わるたびにrenderAll()を呼ぶ設計にします。
ui.js（UI描画モジュール）
// 星評価のHTML生成 const renderStars = (rating, interactive = false, id = "") => { return [1, 2, 3, 4, 5].map(n => ` <span class="star ${{n <= rating ? "filled" : ""}}" ${{interactive ? `data-rating="${n}" data-id="${id}"` : ""}}>★</span> `).join(""); }; // 書籍カード1枚のHTMLを生成 const createBookCard = (book) => ` <article class="book-card" data-id="${{book.id}}"> <div class="book-card-header"> <span class="genre-badge">${{book.genre || "未分類"}}</span> <button class="delete-btn" data-id="${{book.id}}" title="削除">✕</button> </div> <h3 class="book-title">${{book.title}}</h3> <p class="book-author">${{book.author || "著者不明"}}</p> <div class="book-rating">${{renderStars(book.rating)}}</div> ${{book.memo ? `<p class="book-memo">${book.memo}</p>` : ""}} <time class="book-date"> ${{new Date(book.createdAt).toLocaleDateString("ja-JP")}} </time> </article> `; // 書籍一覧を描画 const renderBooks = (books) => { const container = document.querySelector("#book-list"); if (books.length === 0) { container.innerHTML = ` <div class="empty-state"> <p>📚 まだ本が登録されていません</p> <p>最初の1冊を登録してみましょう！</p> </div>`; return; } container.innerHTML = books.map(createBookCard).join(""); }; // 統計情報を更新 const renderStats = ({ total, avg }) => { document.querySelector("#total-count").textContent = `📖 ${{total}}冊`; document.querySelector("#avg-rating").textContent = `⭐ 平均 ${{avg}}`; }; // 全UIを更新（全関数をまとめて呼ぶ） const renderAll = (state) => { const books = getFilteredBooks(state); renderBooks(books); renderStats(getStats()); };
空の状態（Empty State）を必ず実装
UXの基本
データが0件のとき、何も表示しないのは悪いUXです。

✅ 良いEmpty State：
「まだ本が登録されていません」
「最初の1冊を登録してみましょう！」

Empty Stateを用意することで、初めてアプリを使ったユーザーが何をすればいいか分かります。ポートフォリオ作品としての完成度も上がります。

25〜45分
📖
app.jsで全イベントをstateと連携させて、操作のたびにrenderAll(state)を呼ぶ設計を完成させます。
app.js（メイン処理・イベント連携）
// 現在の絞り込み状態 let state = { query: "", genre: "all", sort: "newest" }; // 星評価の入力 let selectedRating = 0; document.querySelector("#rating-input").innerHTML = [1,2,3,4,5].map(n => `<span class="star" data-n="${{n}}">★</span>` ).join(""); document.querySelector("#rating-input").addEventListener("click", (e) => { if (!e.target.dataset.n) return; selectedRating = Number(e.target.dataset.n); document.querySelectorAll("#rating-input .star").forEach((s, i) => s.classList.toggle("filled", i < selectedRating) ); }); // フォームの送信 document.querySelector("#book-form").addEventListener("submit", (e) => { e.preventDefault(); const title = document.querySelector("#title").value; if (!title.trim()) { document.querySelector("#title").classList.add("error"); return; } addBook({ title, author: document.querySelector("#author").value, genre: document.querySelector("#genre").value, rating: selectedRating, memo: document.querySelector("#memo").value, }); e.target.reset(); selectedRating = 0; renderAll(state); }); // 削除（イベントデリゲーション） document.querySelector("#book-list").addEventListener("click", (e) => { if (e.target.classList.contains("delete-btn")) { if (confirm("削除しますか？")) { deleteBook(e.target.dataset.id); renderAll(state); } } }); // フィルター・検索 document.querySelector("#search").addEventListener("input", (e) => { state.query = e.target.value; renderAll(state); }); document.querySelector("#filter-genre").addEventListener("change", (e) => { state.genre = e.target.value; renderAll(state); }); document.querySelector("#sort").addEventListener("change", (e) => { state.sort = e.target.value; renderAll(state); }); // 初期表示 renderAll(state);
