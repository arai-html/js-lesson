script.js（メインの検索処理）
// メインの検索関数 const search = async (city) => { const trimmed = city.trim(); if (!trimmed) { showError("都市名を入力してください"); return; } setLoading(true); try { const raw = await fetchWeather(trimmed); const weather = formatWeather(raw); showWeather(weather); // 検索成功したら都市名をlocalStorageに保存 localStorage.setItem("lastCity", trimmed); addToHistory(trimmed); } catch (error) { showError(error.message); } finally { setLoading(false); } }; // ボタンクリックで検索 el.btn.addEventListener("click", () => search(el.input.value)); // Enterキーでも検索できる el.input.addEventListener("keydown", (e) => { if (e.key === "Enter") search(el.input.value); }); // ページ読み込み時に前回の都市を復元 const lastCity = localStorage.getItem("lastCity"); if (lastCity) { el.input.value = lastCity; search(lastCity); // 前回の都市で自動検索 }
script.js（検索履歴機能）
// 検索履歴を管理（最大5件） let history = JSON.parse(localStorage.getItem("searchHistory")) || []; const addToHistory = (city) => { // 重複を除いて先頭に追加 history = [city, ...history.filter(c => c !== city)].slice(0, 5); localStorage.setItem("searchHistory", JSON.stringify(history)); renderHistory(); }; const renderHistory = () => { const histEl = document.querySelector("#search-history"); if (!histEl) return; histEl.innerHTML = history.map(city => ` <button class="history-btn" data-city="${{city}}">${{city}}</button> `).join(""); }; // 履歴ボタンをクリックすると再検索 document.querySelector("#search-history")?.addEventListener("click", (e) => { if (e.target.classList.contains("history-btn")) { const city = e.target.dataset.city; el.input.value = city; search(city); } }); renderHistory(); // 初期表示
